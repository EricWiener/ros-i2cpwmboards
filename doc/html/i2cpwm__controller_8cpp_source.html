640 void servos_drive (const geometry_msgs::Twist::ConstPtr& msg)
641 {
642  _funcname = _CONST("servos_twist");
643  /* this subscription works on the active_board */
644 
645  int i;
646  int swap;
647  int temp_x, temp_y, temp_r, dir_x, dir_y, dir_r;
648  int speed[4];
649 
650  /* msg is a pointer to a Twist message: msg->linear and msg->angular each of which have members .x .y .z */
651  /* the subscriber uses the maths from: http://robotsforroboticists.com/drive-kinematics/ */
652 
653  ROS_INFO("%s::%s:: [%5.2f %5.2f %5.2f] [%5.2f %5.2f %5.2f]", _pkgname, _funcname,
654  msg->linear.x, msg->linear.y, msg->linear.z, msg->angular.x, msg->angular.y, msg->angular.z);
655 
656  if (_active_drive.mode == MODE_UNDEFINED) {
657  ROS_ERROR("%s::%s:: drive mode not set", _pkgname, _funcname);
658  return;
659  }
660  if ((_active_drive.mode < MODE_UNDEFINED) || (_active_drive.mode >= MODE_INVALID)) {
661  ROS_ERROR("%s::%s:: unrecognized drive mode set %d", _pkgname, _funcname, _active_drive.mode);
662  return;
663  }
664 
665  dir_x = ((msg->linear.x > 0) ? 1 : -1);
666  dir_y = ((msg->linear.y > 0) ? 1 : -1);
667  dir_r = ((msg->angular.z > 0) ? 1 : -1);
668 
669  temp_x = _abs(msg->linear.x);
670  temp_y = _abs(msg->linear.x);
671  temp_r = _abs(msg->angular.z);
672 
673  temp_x = _smoothing (temp_x);
674  temp_y = _smoothing (temp_y);
675  temp_r = _smoothing (temp_r) / 2;
676 
677 
678  switch (_active_drive.mode) {
679  case MODE_ACKERMAN:
680  /*
681  with ackerman drive, steering is handled by a separate servo
682  we drive assigned servos exclusively by the linear.x
683  */
684  speed[0] = temp_x * dir_x;
685  ROS_INFO("%s::%s:: ackerman drive mode speed=%d", _pkgname, _funcname, speed[0]);
686  break;
687  case MODE_DIFFERENTIAL:
688  /*
689  with differential drive, steering is handled by the relative speed of left and right servos
690  we drive assigned servos by mixing linear.x and angular.z with a very simple formula
691  */
692  speed[0] = temp_x * dir_x;
693  speed[1] = (temp_x - temp_r) * dir_x;
694  if (dir_r < 0) {
695  swap = speed[0];
696  speed[0] = speed[1];
697  speed[1] - swap;
698  }
699  ROS_INFO("%s::%s:: differential drive mode speed left=%d right=%d", _pkgname, _funcname, speed[0], speed[1]);
700  break;
701  case MODE_MECANUM:
702  /*
703  with mecanum drive, steering is handled by the relative speed of left and right servos
704  with mecanum drive, lateral motion is handled by the rotation of front and rear servos
705  we drive assigned servos by mixing linear.x and angular.z and linear.y
706  */
707  speed[0] = speed[2] = temp_x * dir_x;
708  speed[1] = speed[3] = (temp_x - temp_r) * dir_x;
709  speed[0] -= temp_y * dir_y;
710  speed[3] -= temp_y * dir_y;
711  speed[1] += temp_y * dir_y;
712  speed[2] += temp_y * dir_y;
713  speed[0] = _absmax (speed[0], 1000);
714  speed[1] = _absmax (speed[1], 1000);
715  speed[2] = _absmax (speed[2], 1000);
716  speed[3] = _absmax (speed[3], 1000);
717  ROS_INFO("%s::%s:: mecanum drive mode speed leftfront=%d rightfront=%d leftrear=%d rightreer=%d", _pkgname, _funcname, speed[0], speed[1], speed[2], speed[3]);
718  break;
719  default:
720  break;
721  }
722 
723  /* find all drive servos and set their new speed */
724  for (i=0; i<16; i++) {
725  if (_servo_configs[_active_board-1][i].mode_pos == POSITION_LEFTFRONT)
726  _set_pwm_interval_proportional (i+1, speed[0]);
727  if (_servo_configs[_active_board-1][i].mode_pos == POSITION_RIGHTFRONT)
728  _set_pwm_interval_proportional (i+1, speed[1]);
729  if (_servo_configs[_active_board-1][i].mode_pos == POSITION_LEFTREAR)
730  _set_pwm_interval_proportional (i+1, speed[3]);
731  if (_servo_configs[_active_board-1][i].mode_pos == POSITION_RIGHTREAR)
732  _set_pwm_interval_proportional (i+1, speed[2]);
733  }
734 }
